# Build image
docker build -t ngelakoni-postgres:latest .

# Build volume
docker volume create ngelakoni-volume

# Build container
docker run -d -p 5433:5432 -v ngelakoni-volume --hostname postgres --name ngelakoni-postgres ngelakoni-postgres:latest

# Cari IP address
docker container inspect ngelakoni-postgres | grep "IPAddress"

# Check data database yang diinput dari app
docker exec -it ngelakoni-postgres bash
psql -U postgres
\l
\c ngelakoni_app
